#### 1.2.3 核心模块职责

| 模块 | 职责说明 |
|:---|:---|
| `controller` | 接收 HTTP 请求，参数校验，调用 Service，返回统一响应格式 `Result<T>` |
| `service` | 核心业务逻辑，事务控制，数据编排，权限校验 |
| `mapper` | 数据库访问，MyBatis-Plus BaseMapper 扩展 |
| `entity` | 数据库实体，与表结构一一对应 |
| `dto/request` | 接口入参，配合 `@Valid` 做参数校验 |
| `dto/response` | 接口出参，控制返回字段，避免暴露敏感信息 |
| `vo` | 复杂查询的聚合结果，如"批次详情 VO"包含批次信息+工人考勤列表 |
| `interceptor` | 管理员/司机双通道 JWT 认证，操作日志自动记录 |

```java
    "code": 200,           // 业务状态码: 200成功, 4xx客户端错误, 5xx服务端错误
    "message": "操作成功",  // 提示信息
    "data": { ... },       // 业务数据
    "timestamp": 1717491600000  // 时间戳
}
```

### 1.4 微信小程序架构设计

|:---|:---|:---|

### 4.2 后端代码规范

```java
/**
 * 审核考勤批次
 * 
 * @param batchId 批次ID
 * @param request 审核请求参数（包含工人考勤修改数据）
 * @return 审核后的批次信息
 * @throws BusinessException 批次不存在或状态不允许审核时抛出
 */
@Transactional(rollbackFor = Exception.class)
public BatchReviewVO reviewBatch(Long batchId, BatchReviewRequest request) {
    // 1. 校验批次是否存在且状态为"待审核"
    AttendanceBatch batch = attendanceBatchMapper.selectById(batchId);
    if (batch == null) {
        throw new BusinessException("批次不存在");
    }
    if (batch.getStatus() != BatchStatus.PENDING.getCode()) {
        throw new BusinessException("只有待审核的批次才能进行审核");
    }
    
    // 2. 校验所有工人考勤记录的项目是否已分配（审核时必须分配项目）
    List<WorkerAttendanceRecord> records = workerAttendanceRecordMapper.selectByBatchId(batchId);
    for (WorkerAttendanceRecord record : records) {
        if (record.getProjectId() == null) {
            throw new BusinessException("工人【" + record.getWorkerName() + "】未分配项目，无法审核");
        }
    }
    
    // 3. 更新批次状态为"已通过"
    // ...
    
    // 4. 计算并写入每个工人的当日工资（避免后续重复计算）
    // ...
    
    // 5. 生成司机当日考勤记录（司机全勤）
    // ...
    
    // 6. 记录操作日志
    // ...
    
    return convertToVO(batch);
```

### 4.3 前端代码规范

```typescript
/**
 * 提交考勤批次
 * 
 * @param params 批次提交参数
 * @returns 提交结果
 */
const submitAttendanceBatch = async (params: BatchSubmitRequest): Promise<Result<BatchInfo>> => {
  // 参数校验
  if (!params.workerList || params.workerList.length === 0) {
    throw new Error('请至少选择一名工人');
  }
  
  // 校验加班时长必须为 0.5 的倍数
  for (const worker of params.workerList) {
    if (worker.overtimeHours % 0.5 !== 0) {
      throw new Error(`工人 ${worker.workerName} 的加班时长必须是 0.5 小时的整数倍`);
    }
  }
  
  // 提交请求
  const response = await api.post('/driver/batches', params);
  return response.data;
};
```
